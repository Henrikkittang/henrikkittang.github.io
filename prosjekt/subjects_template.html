                                def abc_formula(a, b, c):
                                    if b**2 < 4*a*c:
                                        print("no solutions")
                                        return

                                    x1 = (-1 * b + sqrt(b**2 - 4*a*c)) / 2*a
                                    x2 = (-1 * b - sqrt(b**2 - 4*a*c)) / 2*a

                                    return x1, x2


                                print(abc_formula(a1, b1, c1))
